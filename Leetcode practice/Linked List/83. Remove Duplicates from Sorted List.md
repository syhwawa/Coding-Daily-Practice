Given a sorted linked list, delete all duplicates such that each element appear only once.

Example 1:

Input: 1->1->2
Output: 1->2
Example 2:

Input: 1->1->2->3->3
Output: 1->2->3

```Python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def deleteDuplicates(self, head: ListNode) -> ListNode:
        p1 = head
        p2 = head
        while p2:
            if p1.val == p2.val:
                p1.next = p2.next
                p2 = p2.next
            else:
                p1 = p1.next
                p2 = p2.next
        
        return head
````
